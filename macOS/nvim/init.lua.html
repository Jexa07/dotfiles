234 -- Functional wrapper for mapping custom keybindings
235 function map(mode, lhs, rhs, opts)
236     local options = { noremap = true }
237     if opts then
238         options = vim.tbl_extend("force", options, opts)
239     end
240     vim.api.nvim_set_keymap(mode, lhs, rhs, options)
241 end
